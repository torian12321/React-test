States
```
initialState = {val: 50};
<div>
	<Range 
		value    = {state.val}
		onChange = {(e) => setState({val: e})}
	/>

	<ProgressBar val={state.val} type='circular' state='main'/>
	<ProgressBar val={state.val} type='circular' state='sec'/>
	<ProgressBar val={state.val} type='circular' state='success'/>

</div>
```
